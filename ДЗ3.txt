user@user-VirtualBox:~$ sudo groupadd newgroup # добавили группу
user@user-VirtualBox:~$ nano /etc/group # проверяем, группа добавилась
user@user-VirtualBox:~$ sudo usermod -aG newgroup testuser1 # добавили пользователя (а) в группу,  (G) не меняя его основной группы
user@user-VirtualBox:~$ nano /etc/group # проверяем - пользователь в группе
user@user-VirtualBox:~$ sudo addgroup newgroup2

user@user-VirtualBox:~$ sudo usermod -g newgroup2 testuser1
